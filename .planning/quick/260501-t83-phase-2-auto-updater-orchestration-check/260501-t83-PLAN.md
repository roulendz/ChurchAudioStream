```rust
use serde::{Deserialize, Serialize};
use std::fmt;
use std::path::Path;

#[derive(Serialize, Deserialize, Default, Debug, Clone, PartialEq)]
pub struct UpdateState {
    pub last_check_unix: i64,
    pub last_dismissed_unix: i64,
    pub skipped_versions: Vec<String>,
}

#[derive(Debug)]
pub enum StorageError {
    Io(std::io::Error),
    Parse(serde_json::Error),
}

impl fmt::Display for StorageError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            StorageError::Io(error) => write!(f, "update-state IO failed: {error}"),
            StorageError::Parse(error) => write!(f, "update-state JSON parse failed: {error}"),
        }
    }
}

impl std::error::Error for StorageError {}    // empty body — Phase 1 consistency (defer source() until 3rd error type lands)

impl From<std::io::Error> for StorageError {
    fn from(error: std::io::Error) -> Self { StorageError::Io(error) }
}

impl From<serde_json::Error> for StorageError {
    fn from(error: serde_json::Error) -> Self { StorageError::Parse(error) }
}

pub fn load(path: &Path) -> Result<UpdateState, StorageError> {
    let content = match std::fs::read_to_string(path) {
        Ok(content) => content,
        Err(error) if error.kind() == std::io::ErrorKind::NotFound => {
            return Ok(UpdateState::default());
        }
        Err(error) => return Err(StorageError::Io(error)),
    };
    if content.is_empty() {
        return Ok(UpdateState::default());
    }
    let state = serde_json::from_str(&content)?;        // From<serde_json::Error>
    Ok(state)
}

pub fn save(path: &Path, state: &UpdateState) -> Result<(), StorageError> {
    let json = serde_json::to_string_pretty(state)?;     // From<serde_json::Error>
    std::fs::write(path, json)?;                          // From<std::io::Error>
    Ok(())
}

pub fn with_dismissed_now(mut state: UpdateState, now_unix: i64) -> UpdateState {
    state.last_dismissed_unix = now_unix;
    state
}

pub fn with_skipped_version(mut state: UpdateState, version: &str) -> UpdateState {
    if state.skipped_versions.iter().any(|v| v == version) {
        return state;                                      // dedupe — no-op
    }
    state.skipped_versions.push(version.to_string());
    state
}

pub fn with_check_completed(mut state: UpdateState, now_unix: i64) -> UpdateState {
    state.last_check_unix = now_unix;
    state
}
```

Notes:
- `mut state` parameter binding (NOT `state: UpdateState` followed by `let mut state = state;`) is the idiomatic by-value-then-mutate pattern. Caller's value is moved in; helper returns the new owned state.
- `with_skipped_version` test `with_skipped_version_dedupes` must verify a second push of the same string does NOT grow the vec.
- DO NOT use `to_writer` / `File::create` / `flush` chains — `to_string_pretty + fs::write` is one error path per call.

- `serde_json::from_str("")` returns Err; the explicit `content.is_empty()` early-return handles it cleanly without parse-error branching.

```rust
//! Pure decision functions for the auto-updater orchestrator.
//!
//! Zero IO, zero Tauri imports, zero clock calls (now_unix injected by caller).
//! All functions infallible at the boundary — bad input folds into
//! `UpdateDecision::SilentSkip(reason)` strings, never panics, never returns Err.

use crate::update::manifest::{asset_for_platform, UpdateManifest};
use crate::update::version::parse_semver;

/// Outcome of `evaluate_update`. `Notify` carries everything Phase 3's IPC layer
/// needs to construct an `UpdateAvailablePayload` without reaching back into the
/// manifest. `SilentSkip(reason)` reasons match plan strings exactly.
#[derive(Debug, PartialEq)]
pub enum UpdateDecision {
    Notify {
        version: String,
        notes: String,
        platform_url: String,
    },
    SilentSkip(String),
    NoUpdate,
}

/// True iff at least `min_interval_seconds` have elapsed since `last_check_unix`.
///
/// Tiger-Style: `debug_assert!(now_unix >= last_check_unix)` — caller contract
/// is monotonic clock. The runtime `if now_unix < last_check_unix` branch is
/// defense in depth for production clock skew (VM suspend/resume, NTP rollback,
/// user manually adjusts system clock). Returning `true` heals the skew the
/// next call updates `last_check_unix`. Returning `false` would freeze checks.
pub fn should_check_now(
    last_check_unix: i64,
    now_unix: i64,
    min_interval_seconds: i64,
) -> bool {
    debug_assert!(
        now_unix >= last_check_unix,
        "now_unix ({now_unix}) must be >= last_check_unix ({last_check_unix})"
    );
    debug_assert!(min_interval_seconds >= 0);

    if now_unix < last_check_unix {
        return true;
    }
    now_unix - last_check_unix >= min_interval_seconds
}

/// True iff `version` appears in the `skipped` list. One-liner per plan signature.
pub fn is_version_skipped(version: &str, skipped: &[String]) -> bool {
    skipped.iter().any(|v| v == version)
}

/// Top-level decision: should we surface this update to the user?
///
/// Combines version comparison + skip list + dismiss cooldown + per-platform asset
/// availability. Returns `Notify` only after every guard passes. Flat early-return
/// chain — no nested ifs (plan §3 Phase 2 GOOD template). All bad-input cases fold
/// into `SilentSkip(reason)` so the caller has one branch per outcome.
///
/// 7-param signature (A1 resolution): `platform_key` is required to look up the
/// per-platform download URL inside `Notify`. Phase 3 passes the result of
/// `tauri_plugin_os::platform()` (or equivalent).
///
/// Missing-asset behavior (A2 resolution): if `asset_for_platform` returns `None`
/// AT THE Notify-construction step, returns `SilentSkip("no asset for platform {key}")`
/// — fail-fast Tiger-Style. Empty `download_url` strings would be silently broken.
pub fn evaluate_update(
    current: &str,
    manifest: &UpdateManifest,
    platform_key: &str,
    skipped: &[String],
    last_dismissed_unix: i64,
    now_unix: i64,
    dismiss_cooldown_seconds: i64,
) -> UpdateDecision {
    let latest = match parse_semver(&manifest.version) {
        Err(error) => return UpdateDecision::SilentSkip(format!("bad manifest: {error}")),
        Ok(value) => value,
    };
    let current_parsed = match parse_semver(current) {
        Err(error) => return UpdateDecision::SilentSkip(format!("bad current: {error}")),
        Ok(value) => value,
    };
    if latest <= current_parsed {
        return UpdateDecision::NoUpdate;
    }
    if is_version_skipped(&manifest.version, skipped) {
        return UpdateDecision::SilentSkip("user skipped".to_string());
    }
    if now_unix - last_dismissed_unix < dismiss_cooldown_seconds {
        return UpdateDecision::SilentSkip("dismissed cooldown".to_string());
    }
    let asset = match asset_for_platform(manifest, platform_key) {
        None => return UpdateDecision::SilentSkip(format!("no asset for platform {platform_key}")),
        Some(asset) => asset,
    };
    UpdateDecision::Notify {
        version: manifest.version.clone(),
        notes: manifest.notes.clone(),
        platform_url: asset.url.clone(),
    }
}
```

   - ACTUALLY, a simpler and more robust approach: Just use the module-level flag. On cleanup, DO clean up and reset the flag.            });
     ```typescript
           const unlistenError = await listen<string>("sidecar-error", (event) => {
             if (!aborted) addLogEntry(event.payload);
           });
           if (aborted) {
             // Cleanup was called while we were setting up
             unlistenLog();
             unlistenError();
             return;
           }
           cleanupRef.current = [unlistenLog, unlistenError];
         };
         setupListeners().catch((error) => {
           console.warn(

         });

       };
     ```

**src-tauri/src/lib.rs:**

1. Add a shared log buffer using `Arc<Mutex<Vec<String>>>`:
   ```rust
   use std::sync::Mutex;

   // Buffer to store early sidecar logs before frontend registers listeners
   struct LogBuffer {
       entries: Vec<String>,
       capacity: usize,
   }

   impl LogBuffer {
       fn new(capacity: usize) -> Self {
           Self { entries: Vec::with_capacity(capacity), capacity }
       }

       fn push(&mut self, entry: String) {
           if self.entries.len() >= self.capacity {
               self.entries.remove(0); // Drop oldest if full
           }
           self.entries.push(entry);
       }

       fn drain(&mut self) -> Vec<String> {
           std::mem::take(&mut self.entries)
       }
   }
   ```

2. Create a Tauri managed state for the log buffer:
   - Define `struct AppLogBuffer(Mutex<LogBuffer>);`
   - In the `run()` function, before `tauri::Builder`, create: `let log_buffer = AppLogBuffer(Mutex::new(LogBuffer::new(500)));`
   - Pass it to Tauri via `.manage(log_buffer)` on the Builder.

3. In `spawn_sidecar`, access the log buffer via `app_handle.state::<AppLogBuffer>()` within the async spawn block (idiomatic Tauri managed state access — no function signature change needed). In the stdout/stderr handlers:
   - BEFORE emitting the event, also push the log line into the buffer:
     ```rust
     CommandEvent::Stdout(line_bytes) => {
         let log_line = String::from_utf8_lossy(&line_bytes).to_string();
         if let Ok(mut buffer) = log_buffer_handle.lock() {
             buffer.push(log_line.clone());
         }
         let _ = app_handle.emit("sidecar-log", &log_line);
     }
     ```
   - Same for Stderr (push before emit).

4. Add a Tauri command `get_buffered_logs` that drains and returns the buffer:

   ```rust
   #[tauri::command]
   fn get_buffered_logs(buffer: tauri::State<'_, AppLogBuffer>) -> Vec<String> {
       buffer.0.lock().map(|mut b| b.drain()).unwrap_or_default()
   }
   ```

5. Register the command in the Builder: `.invoke_handler(tauri::generate_handler![get_buffered_logs])`

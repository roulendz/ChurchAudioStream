    **Create `sidecar/src/streaming/streaming-types.ts`:**

    Define these types (informed by research code examples):

    ```typescript
    import type { types as mediasoupTypes } from "mediasoup";

    /** Latency mode for a channel. */
    export type LatencyMode = "live" | "stable";

    /** Loss recovery strategy for a channel. */
    export type LossRecoveryMode = "nack" | "plc";

    /** State tracked per mediasoup Worker. */
    export interface WorkerState {
      readonly worker: mediasoupTypes.Worker;
      readonly index: number;
      memoryMonitorInterval: NodeJS.Timeout | null;
    }

    /** Streaming state for a single channel (Router + PlainTransport + Producer). */
    export interface ChannelStreamingState {
      readonly channelId: string;
      readonly router: mediasoupTypes.Router;
      readonly plainTransport: mediasoupTypes.PlainTransport;
      readonly audioProducer: mediasoupTypes.Producer;
    }

    /** Data attached to each protoo Peer for a listener connection. */
    export interface ListenerPeerData {
      readonly sessionId: string;
      readonly connectedAt: number;
      rtpCapabilities: mediasoupTypes.RtpCapabilities | null;
      webRtcTransport: mediasoupTypes.WebRtcTransport | null;
      currentConsumer: mediasoupTypes.Consumer | null;
      currentChannelId: string | null;
      isAdmin: boolean;
    }

    /** Summary of a listener connection for admin dashboard. */
    export interface ListenerSessionInfo {
      readonly sessionId: string;
      readonly connectedAt: number;
      readonly channelId: string | null;
      readonly isAdmin: boolean;
    }

    /** Channel info pushed to listeners (subset of AppChannel). */
    export interface ListenerChannelInfo {
      readonly id: string;
      readonly name: string;
      readonly language?: string;
      readonly description?: string;
      readonly isDefault: boolean;
    }

    /** Per-channel latency estimation components. */
    export interface LatencyEstimate {
      readonly gstreamerBufferMs: number;
      readonly opusEncodeMs: number;
      readonly mediasoupProcessingMs: number;
      readonly webrtcJitterBufferMs: number;
      readonly totalMs: number;
    }

    /** Worker resource usage snapshot for admin dashboard. */
    export interface WorkerResourceInfo {
      readonly index: number;
      readonly pid: number;
      readonly memoryMb: number;
      readonly cpuPercent: number;
    }

    /** Opus RTP codec parameters matching Phase 3 pipeline builder. */
    export const OPUS_RTP_CODEC: mediasoupTypes.RtpCodecCapability = {
      kind: "audio",
      mimeType: "audio/opus",
      clockRate: 48000,
      channels: 2,
    };

    /** Payload type matching rtpopuspay pt=101 in pipeline-builder.ts. */
    export const OPUS_PAYLOAD_TYPE = 101;
    ```
